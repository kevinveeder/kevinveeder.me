NBA Leaders

# OFFENSE

url <- "https://www.espn.com/nba/stats"
nbaleaders_web <- read_html(url)
nbaleaders_tables <- html_table(nbaleaders_web)
# table 1
leader_pts <- as.data.frame(nbaleaders_tables[[1]])
leader_pts$Points <- str_replace(leader_pts$Points, pattern = '[[:digit:]]', replace = "")
leader_pts$Points <- leader_pts$Points %>%  str_sub(start = 1L, end = -4L)
leader_pts <- leader_pts[-6,]
leader_pts$PTS <- as.numeric(leader_pts$PTS)
colnames(leader_pts) <- c("Player", "Pts")
ggplot(leader_pts, aes(y = Pts)) + geom_point(aes(x = 1:5, color = Player)) + labs(x = "Rank", y = "Points Per Game", title = "NBA Point Scorers")

NFL Leaders

# Rushing
nfl_url <- "https://www.espn.com/nfl/stats/player/_/stat/rushing"
nfl_web <- read_html(nfl_url)
nfl_leaders.table <- html_table(nfl_web)
rushing_leaders.df <- data.frame(nfl_leaders.table[[1]], nfl_leaders.table[[2]])
colnames(rushing_leaders.df)
##  [1] "RK"    "Name"  "POS"   "GP"    "ATT"   "YDS"   "AVG"   "LNG"  
##  [9] "BIG"   "TD"    "YDS.G" "FUM"   "LST"   "FD"
rushing_leaders.df$RK <- as.numeric(rushing_leaders.df$RK)
rushing_leaders.df$Name <- factor(rushing_leaders.df$Name)
rushing_leaders.df$POS <- factor(rushing_leaders.df$POS)
rushing_leaders.df$ATT <- as.numeric(rushing_leaders.df$ATT)
rushing_leaders.df$TD <- as.numeric(rushing_leaders.df$TD)

rushing_leaders.df %>% 
  select(RK, Name, POS, ATT, TD, YDS.G, FUM) %>% 
  filter(RK <= 5) %>% 
  ggplot(aes(y = YDS.G)) + geom_point(aes(x = 1:5, color = Name, size = 5:1)) + labs(x = "Rank", y = "Yards per game", title = "NFL Rushing Leaders")
